1. Cập nhật main.dart (Nâng cấp Bottom Navigation)
Thêm tab Vận động (Fitness) vào thanh điều hướng chính, tinh chỉnh màu sắc sang trọng hơn và tối ưu nút Camera nổi ở giữa.

Dart
import 'package:flutter/material.dart';
import 'package:google_fonts/google_fonts.dart';
import 'screens/dashboard_screen.dart';
import 'screens/camera_screen.dart';
import 'screens/history_screen.dart';
import 'screens/fitness_screen.dart'; // Thêm import này
import 'models/food_item.dart';

void main() {
  runApp(const VinNutriApp());
}

class VinNutriApp extends StatelessWidget {
  const VinNutriApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: 'VinNutri AI',
      theme: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: const Color(0xFF8CE3BE)),
        textTheme: GoogleFonts.outfitTextTheme(),
        scaffoldBackgroundColor: const Color(0xFFF8FAF9), // Màu nền sáng dịu mắt
        useMaterial3: true,
      ),
      home: const MainLayout(),
    );
  }
}

class MainLayout extends StatefulWidget {
  const MainLayout({super.key});

  @override
  State<MainLayout> createState() => _MainLayoutState();
}

class _MainLayoutState extends State<MainLayout> {
  int _selectedIndex = 0; // Bắt đầu ở Dashboard
  final GlobalKey<FoodSnapperScreenState> _cameraKey = GlobalKey();
  
  final List<FoodItem> _history = [
    FoodItem(name: 'Phở Bò', calories: 450, protein: 25, fat: 12, carbs: 60, weight: 650, dateTime: DateTime.now().subtract(const Duration(hours: 5))),
  ];

  void _addFood(FoodItem item) {
    setState(() {
      _history.insert(0, item);
      _selectedIndex = 0; // Chuyển về Home sau khi chụp
    });
    
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
        content: Text('Đã lưu ${item.name} vào nhật ký!'), 
        backgroundColor: const Color(0xFF2D312F),
        behavior: SnackBarBehavior.floating,
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(10)),
      ),
    );
  }

  void _onItemTapped(int index) {
    setState(() {
      _selectedIndex = index;
    });
  }

  @override
  Widget build(BuildContext context) {
    // 4 Màn hình chính
    final List<Widget> screens = [
      DashboardScreen(history: _history),
      const FitnessScreen(), // Màn hình đếm bước
      HistoryScreen(history: _history),
      const Center(child: Text("Cài đặt cá nhân", style: TextStyle(color: Colors.grey))), // Placeholder Cài đặt
    ];

    return Scaffold(
      body: IndexedStack(
        index: _selectedIndex > 1 && _selectedIndex < 4 ? _selectedIndex - 1 : _selectedIndex, // Xử lý index cho Camera
        children: [
          DashboardScreen(history: _history),
          const FitnessScreen(),
          FoodSnapperScreen(key: _cameraKey, onAdd: _addFood), // Index ẩn của Camera
          HistoryScreen(history: _history),
          const Center(child: Text("Cài đặt cá nhân")), 
        ],
      ),
      
      // Nút Chụp Ảnh Nổi (Center Docked)
      floatingActionButton: Container(
        height: 64,
        width: 64,
        margin: const EdgeInsets.only(top: 30),
        child: FloatingActionButton(
          onPressed: () {
            if (_selectedIndex == 2) {
               _cameraKey.currentState?.handleRefresh();
            } else {
              _onItemTapped(2); // Giả sử Index 2 là Camera
            }
          },
          backgroundColor: const Color(0xFF8CE3BE),
          elevation: 8,
          shadowColor: const Color(0xFF8CE3BE).withOpacity(0.5),
          shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(20)),
          child: const Icon(Icons.document_scanner_rounded, color: Colors.white, size: 30),
        ),
      ),
      floatingActionButtonLocation: FloatingActionButtonLocation.centerDocked,
      
      bottomNavigationBar: Container(
        decoration: BoxDecoration(
          boxShadow: [
            BoxShadow(color: Colors.black.withOpacity(0.05), blurRadius: 20, offset: const Offset(0, -5)),
          ],
        ),
        child: BottomAppBar(
          shape: const CircularNotchedRectangle(),
          notchMargin: 10.0,
          color: Colors.white,
          elevation: 0,
          child: SizedBox(
            height: 60,
            child: Row(
              mainAxisAlignment: MainAxisAlignment.spaceBetween,
              children: [
                Row(
                  children: [
                    _buildNavItem(icon: Icons.space_dashboard_rounded, label: 'Home', index: 0),
                    _buildNavItem(icon: Icons.directions_walk_rounded, label: 'Vận động', index: 1),
                  ],
                ),
                Row(
                  children: [
                    _buildNavItem(icon: Icons.history_rounded, label: 'Lịch sử', index: 3),
                    _buildNavItem(icon: Icons.person_rounded, label: 'Hồ sơ', index: 4),
                  ],
                ),
              ],
            ),
          ),
        ),
      ),
    );
  }

  Widget _buildNavItem({required IconData icon, required String label, required int index}) {
    // Logic chọn index hơi đặc biệt vì vướng nút Camera ở giữa
    final isSelected = _selectedIndex == index;
    final color = isSelected ? const Color(0xFF8CE3BE) : const Color(0xFFB0B8B4);

    return InkWell(
      onTap: () => _onItemTapped(index),
      splashColor: Colors.transparent,
      highlightColor: Colors.transparent,
      child: SizedBox(
        width: MediaQuery.of(context).size.width / 5,
        child: Column(
          mainAxisSize: MainAxisSize.min,
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            AnimatedContainer(
              duration: const Duration(milliseconds: 200),
              curve: Curves.easeInOut,
              padding: EdgeInsets.all(isSelected ? 6 : 0),
              decoration: BoxDecoration(
                color: isSelected ? const Color(0xFF8CE3BE).withOpacity(0.15) : Colors.transparent,
                borderRadius: BorderRadius.circular(12),
              ),
              child: Icon(icon, color: color, size: isSelected ? 26 : 24),
            ),
            const SizedBox(height: 4),
            Text(
              label,
              style: GoogleFonts.outfit(
                color: isSelected ? const Color(0xFF2D312F) : const Color(0xFFB0B8B4),
                fontSize: 11,
                fontWeight: isSelected ? FontWeight.w700 : FontWeight.w500,
              ),
            ),
          ],
        ),
      ),
    );
  }
}
2. Làm mới hoàn toàn fitness_screen.dart
Giao diện được làm lại với tỷ lệ chuẩn, thiết kế thẻ (Card) đổ bóng mềm, vòng ring đếm bước có viền bo tròn (StrokeCap.round) và tự động cập nhật số liệu Km dựa trên số bước.

Dart
import 'package:flutter/material.dart';
import 'package:google_fonts/google_fonts.dart';

class FitnessScreen extends StatefulWidget {
  const FitnessScreen({super.key});

  @override
  State<FitnessScreen> createState() => _FitnessScreenState();
}

class _FitnessScreenState extends State<FitnessScreen> with SingleTickerProviderStateMixin {
  int _steps = 2450;
  final int _stepGoal = 10000;
  bool _isTracking = false;

  // Công thức: 1 bước chân trung bình khoảng 0.000762 km (76.2 cm)
  double get _distance => (_steps * 0.000762);
  // Công thức ước tính: 1 bước tốn khoảng 0.04 calo
  int get _caloriesBurned => (_steps * 0.04).toInt();
  int _activeMinutes = 24;

  void _toggleTracking() {
    setState(() {
      _isTracking = !_isTracking;
    });
    if (_isTracking) {
       _startMockTracking();
    }
  }

  void _startMockTracking() async {
    while (_isTracking) {
      await Future.delayed(const Duration(milliseconds: 800));
      if (!mounted) break;
      setState(() {
        _steps += 2; // Tăng 2 bước mỗi nhịp đi bộ
        if (_steps % 100 == 0) _activeMinutes++; // Cứ 100 bước tăng 1 phút (giả lập)
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFFF8FAF9),
      appBar: AppBar(
        backgroundColor: Colors.transparent,
        elevation: 0,
        title: Text(
          "Hoạt động hôm nay", 
          style: GoogleFonts.outfit(color: const Color(0xFF2D312F), fontSize: 24, fontWeight: FontWeight.bold)
        ),
        centerTitle: false,
      ),
      body: SafeArea(
        child: Padding(
          padding: const EdgeInsets.symmetric(horizontal: 24.0),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.center,
            children: [
              const SizedBox(height: 20),
              _buildProgressRing(),
              const SizedBox(height: 40),
              _buildMetricsGrid(),
              const Spacer(),
              _buildActionCard(),
              const SizedBox(height: 30),
            ],
          ),
        ),
      ),
    );
  }

  Widget _buildProgressRing() {
    double progress = _steps / _stepGoal;
    if (progress > 1.0) progress = 1.0;

    return Center(
      child: Container(
        width: 260,
        height: 260,
        decoration: BoxDecoration(
          shape: BoxShape.circle,
          color: Colors.white,
          boxShadow: [
            BoxShadow(color: const Color(0xFF8CE3BE).withOpacity(0.15), blurRadius: 30, spreadRadius: 10),
          ],
        ),
        child: Stack(
          alignment: Alignment.center,
          children: [
            // Background Ring
            SizedBox(
              width: 230,
              height: 230,
              child: CircularProgressIndicator(
                value: 1.0,
                strokeWidth: 16,
                color: const Color(0xFFF0F4F2),
              ),
            ),
            // Progress Ring
            SizedBox(
              width: 230,
              height: 230,
              child: CircularProgressIndicator(
                value: progress,
                strokeWidth: 16,
                backgroundColor: Colors.transparent,
                color: const Color(0xFF8CE3BE),
                strokeCap: StrokeCap.round, // Bo tròn đầu thanh trượt (Cực kỳ quan trọng cho UI xịn)
              ),
            ),
            // Inner Content
            Column(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                const Icon(Icons.directions_walk_rounded, color: Color(0xFF8CE3BE), size: 36),
                const SizedBox(height: 8),
                Text(
                  "$_steps", 
                  style: GoogleFonts.outfit(fontSize: 48, fontWeight: FontWeight.w900, color: const Color(0xFF2D312F), height: 1)
                ),
                Text(
                  "/ $_stepGoal bước", 
                  style: GoogleFonts.outfit(color: const Color(0xFF757D7A), fontSize: 16, fontWeight: FontWeight.w500)
                ),
              ],
            )
          ],
        ),
      ),
    );
  }

  Widget _buildMetricsGrid() {
    return Row(
      mainAxisAlignment: MainAxisAlignment.spaceBetween,
      children: [
        _buildMetricCard(Icons.local_fire_department_rounded, "$_caloriesBurned", "Kcal", const Color(0xFFFF8A65)),
        _buildMetricCard(Icons.map_rounded, _distance.toStringAsFixed(2), "Km", const Color(0xFF4FC3F7)),
        _buildMetricCard(Icons.timer_rounded, "$_activeMinutes", "Phút", const Color(0xFFBA68C8)),
      ],
    );
  }

  Widget _buildMetricCard(IconData icon, String value, String unit, Color color) {
    return Container(
      width: MediaQuery.of(context).size.width * 0.26,
      padding: const EdgeInsets.symmetric(vertical: 20, horizontal: 12),
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(20),
        boxShadow: [
          BoxShadow(color: Colors.black.withOpacity(0.03), blurRadius: 10, offset: const Offset(0, 4)),
        ],
      ),
      child: Column(
        children: [
          Container(
            padding: const EdgeInsets.all(10),
            decoration: BoxDecoration(color: color.withOpacity(0.15), shape: BoxShape.circle),
            child: Icon(icon, color: color, size: 26),
          ),
          const SizedBox(height: 16),
          Text(value, style: GoogleFonts.outfit(fontSize: 22, fontWeight: FontWeight.w800, color: const Color(0xFF2D312F))),
          const SizedBox(height: 2),
          Text(unit, style: GoogleFonts.outfit(color: const Color(0xFF757D7A), fontSize: 13, fontWeight: FontWeight.w500)),
        ],
      ),
    );
  }

  Widget _buildActionCard() {
    return AnimatedContainer(
      duration: const Duration(milliseconds: 300),
      width: double.infinity,
      height: 68,
      decoration: BoxDecoration(
        borderRadius: BorderRadius.circular(24),
        boxShadow: [
          BoxShadow(
            color: _isTracking ? const Color(0xFFFF8A65).withOpacity(0.3) : const Color(0xFF8CE3BE).withOpacity(0.4),
            blurRadius: 15, offset: const Offset(0, 5)
          )
        ],
      ),
      child: ElevatedButton(
        onPressed: _toggleTracking,
        style: ElevatedButton.styleFrom(
          backgroundColor: _isTracking ? const Color(0xFFFF8A65) : const Color(0xFF8CE3BE),
          foregroundColor: Colors.white,
          shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(24)),
          elevation: 0,
        ),
        child: Row(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Icon(_isTracking ? Icons.pause_rounded : Icons.play_arrow_rounded, size: 28),
            const SizedBox(width: 8),
            Text(
              _isTracking ? "TẠM DỪNG ĐI BỘ" : "BẮT ĐẦU ĐI BỘ",
              style: GoogleFonts.outfit(fontSize: 18, fontWeight: FontWeight.bold, letterSpacing: 1),
            ),
          ],
        ),
      ),
    );
  }
}
